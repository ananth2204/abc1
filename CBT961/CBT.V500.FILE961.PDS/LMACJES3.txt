* Mains
sdsf *i main=all
sdsf *i main=all,x

* inquiry and change device
sdsf *i d d=0E00
sdsf *i d d=0E00,sy1
sdsf *i d d=sy1
sdsf *F V 0E00,ONLINE
sdsf *v sy1,off
sdsf *v sy1,on

* inquiry setup
sdsf *i s
* inquiry setup allocation
sdsf *i s a
sdsf *i s a,J=64420
sdsf *i s de=yves.data.set
sdsf *i s v=vol???,e
sdsf *i s,a,summ
* inquiry setup error
sdsf *i s e
sdsf *i s e,J=62966
sdsf *x display,j=64440
sdsf d u,vol=vol001

* inquiry GMS
*             all groups
sdsf *i g sy1 g
*             group DFLT
sdsf *i g sy1 g dflt
*             all classes
sdsf *i g sy1 c
*             class Y
sdsf *i g sy1 c y
sdsf *i c=y

* modify GMS
*             group DFLT OFF
sdsf *f g sy1 g dflt off
*             group DFLT Init change
sdsf *f g sy1 g dflt init 5
*             class Y OFF
sdsf *f g sy1 c y off

* job queues
sdsf *i b
sdsf *i b,sy1
sdsf *i b,g=dflt
sdsf *i q,g=dbms
sdsf *i q,g=dflt

* job
sdsf *i j=1245
sdsf *i j=jobn*
sdsf *i j=j?bn*
sdsf *i j=test*,w,x
sdsf *x display,j=1245
* job active
sdsf *i a
sdsf *i a,sy1
sdsf *i a,g=dflt
* job changes
*              new Class
sdsf *f j=????,C=?
*              new Service Class
sdsf *f j=????,SRVCLASS=?
*              Run a job wainting on a WLM Inititor
sdsf *f j=????,RUN
*              Put a message in the joblog
sdsf *f j=64440,MSG='Das ist lustig'
* job cancel
sdsf *f j=?????,C
sdsf *f j=?????,C,DUMP
* job cancel P=Print O=Output
sdsf *f j=?????,CP
sdsf *f j=?????,CO

* spool
sdsf *i q dd=all
sdsf *i q s
*                Top usage
sdsf *i q sp=all,U

* output
sdsf *i u cl=?
sdsf *i u q=wtr,cl=?
sdsf *i u q=hold,cl=?
sdsf *i u q=hold,cl=?,j=?,n=ALL
sdsf *i u q=hold,cl=t,j=63145,req=all,dd=?

* PROCLIB
sdsf *i proclib

* DJC
sdsf *I,N,ID=DJCNET,LIST
