//TSBATCH  JOB (@SYSUID,@SYSACCT),'LIBSPOOL ACF RULES',CLASS=A
/*JOBPARM Q=T,I
//*
//* LIBSPOOL DATA SETS
//*
//*       1) SYSX.LIBSPOOL.LIBPROD - CURRENT LIBRARIAN JCL LIBRARY
//*                       CONTAINS THE CURRENT JCL MODULES
//*                       GOTTEN FROM THE LIBSPOOL TASK.
//*
//*       2) SYSX.LIBSPOOL.LIBREST - RESTORE FROM LIBRARIAN ARCHIVE
//*                          TAPE LIBRARIAN JCL LIBRARY
//*
//*       3) SYSX.LIBSPOOL.ARCPDS - ARCHIVE DIRECTORY
//*                          PARTIONED DATA SET CONTAINING INFORMATION
//*                          ON THE CONTENTS OF THE LIBRARIAN ARCHIVE
//*                          TAPES.
//*
//*       4) PPODOPR.LIBSPOOL.DALYARCH - DAILY LIBRARIAN ARCHIVE TAPE
//*
//*       5) PPODOPR.LIBSPOOL.WKLYARCH - WEEKLY LIBRARIAN ARCHIVE TAPE
//*
//ACFRULES EXEC  TSBATCH
//TERMINAL DD *
 ACFN K(SYSX)    ADD(LIBSPOOL.LIBPROD UID(SHCS***) +
                 R(A) E(A)) NOLIST
 ACFN K(SYSX)    ADD(LIBSPOOL.LIBPROD UID(SYSX***) +
                 R(A) E(A)) NOLIST
 ACFN K(SYSX)    ADD(LIBSPOOL.LIBPROD UID(SHOP***) +
                 R(A) E(A)) NOLIST
 ACFN K(SYSX)    ADD(LIBSPOOL.LIBPROD UID(PPODOPR) +
                 R(A) W(A) A(A) E(A)) NOLIST
 ACFN K(SYSX)    ADD(LIBSPOOL.ARCPDS UID(SHCS***) +
                 R(A) E(A)) NOLIST
 ACFN K(SYSX)    ADD(LIBSPOOL.ARCPDS UID(SYSX***) +
                 R(A) E(A)) NOLIST
 ACFN K(SYSX)    ADD(LIBSPOOL.ARCPDS UID(SHOP***) +
                 R(A) E(A)) NOLIST
 ACFN K(SYSX)    ADD(LIBSPOOL.ARCPDS UID(PPODOPR) +
                 R(A) W(A) A(A) E(A)) NOLIST
 ACFN K(SYSX)    ADD(LIBSPOOL.LIBREST UID(SHCS***) +
                 R(A) W(A) E(A)) NOLIST
 ACFN K(SYSX)    ADD(LIBSPOOL.LIBREST UID(SYSX***) +
                 R(A) W(A) E(A)) NOLIST
 ACFN K(SYSX)    ADD(LIBSPOOL.LIBREST UID(SHOP***) +
                 R(A) W(A) E(A)) NOLIST
 ACFN K(SYSX)    ADD(LIBSPOOL.LIBREST UID(PPODOPR) +
                 R(A) W(A) A(A) E(A))
 ACFN K(PPODOPR) ADD(LIBSPOOL.DALYARCH.G****V- -
                 UID(SHCS***) R(A) E(A)) NOLIST
 ACFN K(PPODOPR) ADD(LIBSPOOL.DALYARCH.G****V- -
                 UID(SYSX***) R(A) E(A)) NOLIST
 ACFN K(PPODOPR) ADD(LIBSPOOL.DALYARCH.G****V- -
                 UID(SHOP***) R(A) E(A)) NOLIST
 ACFN K(PPODOPR) ADD(LIBSPOOL.WKLYARCH.G****V- -
                 UID(SHCS***) R(A) E(A)) NOLIST
 ACFN K(PPODOPR) ADD(LIBSPOOL.WKLYARCH.G****V- -
                 UID(SYSX***) R(A) E(A)) NOLIST
 ACFN K(PPODOPR) ADD(LIBSPOOL.WKLYARCH.G****V- -
                 UID(SHOP***) R(A) E(A))
