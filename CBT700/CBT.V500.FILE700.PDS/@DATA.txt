/**************************** REXX ************************************/
/*                                                                    */
/*      THIS REXX SUBROUTINE WILL RETURN THE DATA THAT IS ON THE      */
/*      LINE THE CURSOR IS ON AND WILL STORE IT IN THE FIELD CALLED   */
/*      DATALINE.                                                     */
/*                                                                    */
/*    (C) COPYRIGHT JAMES HAIRE                                       */
/*        ALL RIGHTS RESERVED      1995                               */
/*                                                                    */
/*      OUTPUT:    DATALINE - DATA ON THE LINE THAT THE CURSOR IS ON. */
/*                                                                    */
/**********************************************************************/
"ISREDIT (LINEPOS,COLPOS) = CURSOR"
"ISREDIT (DATALINE) = LINE &LINEPOS"
RETURN DATALINE
