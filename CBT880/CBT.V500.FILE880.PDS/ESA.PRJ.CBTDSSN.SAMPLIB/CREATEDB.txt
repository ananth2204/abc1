CREATE TABLE MVSDSSN.DSSJOBS
(
    KEY          TIMESTAMP NOT NULL,
    JOBNAME      CHAR(8),
    JOBNUMBER    CHAR(8),
    PROGRAM      CHAR(8),
    DATASET      CHAR(128),
    JOBLOG       VARCHAR(32000),
    PRIMARY KEY(KEY)
) IN MVSDSSN.DSSNTBLS;
CREATE TYPE 2 INDEX
 XJOBNAME ON MVSDSSN.DSSJOBS
 (JOBNAME ASC);
CREATE TYPE 2 UNIQUE INDEX
 XKEY ON MVSDSSN.DSSJOBS
 (KEY ASC) CLUSTER;
CREATE TABLE MVSDSSN.DSSDATASETS
(
    KEY          TIMESTAMP NOT NULL,
    JOB          TIMESTAMP NOT NULL,
    DATASET      CHAR(44),
    VOLUMES      CHAR(36),
    TYPE         INTEGER,
    SIZE         INTEGER,
    PRIMARY KEY(KEY),
    FOREIGN KEY  FJOB (JOB)
    REFERENCES   MVSDSSN.DSSJOBS
) IN MVSDSSN.DSSNTBLS;
CREATE TYPE 2 INDEX
XDATASET ON MVSDSSN.DSSDATASETS
 (DATASET ASC);
CREATE TYPE 2 UNIQUE INDEX
 YKEY ON MVSDSSN.DSSDATASETS
 (KEY ASC) CLUSTER;
