 $HBF:
     PROC(HEX) RETURNS(BIN FIXED(31));
     DCL    HEX     CHAR(*) VAR,
            ALFA    CHAR(16) STATIC INT INIT('0123456789ABCDEF'),
            (I,L)   BIN FIXED,
         1  BIT_STR,
          2 RSRV    BIT(16),
          2 BIT16   BIT(16),
            J       BIN FIXED(31) DEF BIT_STR;
      L=LENGTH(HEX);
      J=0;
      DO I=L TO 1 BY -1;
         K=INDEX(ALFA,SUBSTR(HEX,I,1));
         IF K=0 THEN RETURN(-1);
         J=J+(K-1)*(16**(L-I));
      END;
      RETURN (J);
 END $HBF;
 END REASM$;
