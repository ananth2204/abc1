*PROCESS M,NOINSOURCE,S;
 /*
//PGMVCI JOB 'IBI','V CHERNYAK',REGION=1000K,NOTIFY=PGMVC,CLASS=A,
//        MSGCLASS=A,MSGLEVEL=(1,1)
//PROCLIB  JCLLIB ORDER=PGMVYC.REASM.PLI
//        EXEC PLIXCL,PARM.LKED='LET,NCAL'
//PLI.SYSIN     DD DSN=PGMVYC.REASM.PLI(REASMCR),DISP=SHR
//PLI.SYSLIB    DD DSN=PGMVYC.REASM.PLI,DISP=SHR
//LKED.SYSLMOD  DD DSN=PGMVYC.REASM.SUBLOAD(REASMCR),DISP=SHR
//
 */
 REASMCR:PROC(SCREEN_PTR);
    DCL SCREEN_PTR PTR;
    %INCLUDE SCREEN,REASM99;
    #FORM('REASM99',LOAD);
    SOUND=#YES;
    MODE=#OUTB;
    #CUR(LINE1,2,NAME);
    LINE1='I';
    #CALLTV;
    DELAY(1000*2);
    #CUR(LINE2,4,NAME);
    LINE2=' AM';
    #CALLTV;
    DELAY(1000*2);
    #CUR(LINE3,5,NAME);
    LINE3='VERY';
    #CALLTV;
    DELAY(1000*2);
    #CUR(LINE4,2,NAME);
    LINE4=' ';
    #CALLTV;
    DELAY(1000*2);
    #CUR(LINE5,6,NAME);
    LINE5='TIRED';
    #CALLTV;
    DO I=1 TO 5;
      #CALLTV;
      DELAY(1000*1);
    END;
    STOP;
 END REASMCR;
