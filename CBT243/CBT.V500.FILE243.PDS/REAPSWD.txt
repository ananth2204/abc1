*PROCESS S,OPTIONS;
 /*
//REAPSWD JOB REGION=1000K
// EXEC PLXCM,I1='MVS.REASM.PLI',B1=,M=REAPSWD
//LKED.SYSLIB DD
// DD DSN=MVS.REASM.SUBLOAD,DISP=SHR
//
 */
 REAPSWD: PROC OPTIONS(MAIN);
   DCL   REASQPW ENTRY EXT,
         CPU     CHAR(12),
         DATE    CHAR(6),
         PSWD    CHAR(19);


   DO WHILE('1'B);
      DISPLAY
        ('REASQPW:ENTER CPU NUMBER AND DATE "NNNNNNNNNNNN,YY.DDD"')
        REPLY(PSWD);
      IF PSWD='U' THEN LEAVE;
      GET STRING(PSWD) EDIT(CPU,DATE)(A(12),X(1),A(6));
      IF CPU='************' THEN CPU='';
      PSWD = CPU;
      CALL REASQPW('CRYPT',PSWD,DATE);
      DISPLAY('REASQPW:PASSWORD FOR CPU "' || CPU || ' DATE "' || DATE
              || '"   IS "' || PSWD || '"');
   END;

 END REAPSWD;
