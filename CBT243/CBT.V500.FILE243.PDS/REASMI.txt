 REASMI:
      PROC(@,CMDSET,@COMPT) REORDER;

 %DCL    @I  FIXED;
 %@I=0;
 %@MN:   PROC(N,V,F,P,DUMMY) RETURNS(CHAR);
     DCL (N,V,F,P,DUMMY) CHAR;
     DCL J FIXED;
     DCL (C,VV,RET) CHAR;

     VV = '';
     DO J=1 TO LENGTH(V);
        C = SUBSTR(V,J,1);
        IF C='0' THEN VV = VV || '0000';
        IF C='1' THEN VV = VV || '0001';
        IF C='2' THEN VV = VV || '0010';
        IF C='3' THEN VV = VV || '0011';
        IF C='4' THEN VV = VV || '0100';
        IF C='5' THEN VV = VV || '0101';
        IF C='6' THEN VV = VV || '0110';
        IF C='7' THEN VV = VV || '0111';
        IF C='8' THEN VV = VV || '1000';
        IF C='9' THEN VV = VV || '1001';
        IF C='A' THEN VV = VV || '1010';
        IF C='B' THEN VV = VV || '1011';
        IF C='C' THEN VV = VV || '1100';
        IF C='D' THEN VV = VV || '1101';
        IF C='E' THEN VV = VV || '1110';
        IF C='F' THEN VV = VV || '1111';
     END;

     IF LENGTH(V)<4 THEN
        VV = '00000000' || VV;

     @I=@I+1;
     RET =
      '@('     || @I || ').V='''    || VV ||
      '''B;@(' || @I || ').N='''    || N  ||
      ''';@('  || @I || ').F=$'     || F  ||
      ';@('    || @I || ').PRIV=''' || P  || '''B;';

     RETURN(RET);

 %END @MN;
 %ACT @MN;

 %END;
