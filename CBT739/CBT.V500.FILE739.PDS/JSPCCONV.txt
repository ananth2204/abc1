/* DESCRIPTIVE_NAME := Configuration table conversion routine        */
/*                                                                   */
/* FUNCTION  = Converts configuration table assembler file into      */
/*             keyword format used in ISPF for OS/390 R8 and later.  */
/*                                                                   */
/* NOTES = none                                                      */
/*                                                                   */
/* INPUT := Configuration table assembler source                     */
/*                                                                   */
/* OUTPUT := Configuration table flat file.                          */
/*                                                                   */

Trace o

address ispexec
exec_rc = 0
'VGET (ZCNVSRCE ZCNVSRCM ZCNVKWD ZCNVKWDM ZCNVOPT) PROFILE'
zcnvcsr = 'ZCMD'
call convert
exit exec_rc

/* Convert: Main processing logic **********************************/
convert:
address tso
call initialize_variables
If exec_rc = 0 Then
  call initialize_files
If exec_rc = 0 Then
  call create_header
If exec_rc = 0 Then
  call get_values_from_asm_source
If exec_rc = 0 Then
  call get_default_values
If exec_rc = 0 Then
  call rectify_differences
If exec_rc = 0 Then
  Do
    If (zcnvopt = 'ALL' ) Then
      Do
        aa = lastgroup + 1
        If grouptitle.aa ¬= '' Then
          Do aaa = aa to groups
            call addline '/*'copies('-',68)
            call addline('/*'center(grouptitle.aaa,68)'*/')
            call addline '/*'copies('-',68)
            aa = aa + 1
          End
      End
  End
If exec_rc = 0 Then
  call view_output
x = msg('OFF')
'FREE F(OUTFILE INFILE)'
x = msg('ON')
return

/* Initialize_Files: Allocate input and output and read isrcnfig ***/

initialize_files:
  call msg 'ON'
  /* Allocate assembler input                                        */
  If zcnvsrcm ¬= '' Then
    If substr(zcnvsrce,1,1) = "'" Then
      srcname = "'"strip(zcnvsrce,'B',"'")"("zcnvsrcm")'"
    Else
      srcname = strip(zcnvsrce,'B',"'")"("zcnvsrcm")"
  Else
    srcname = zcnvsrce
  'ALLOC F(INFILE) DA('srcname') SHR REU'
  If rc > 0 Then
    Do
      exec_rc = 8
      Address ISPEXEC 'SETMSG MSG(ISPC269)'
    End
  If exec_rc = 0 Then
    Do
      /* Read assembler file into a stem variable                    */
      'EXECIO * DISKR INFILE (FINIS STEM CONFIG.'
      If rc > 0 Then
        Do
          exec_rc = 8
          Address ISPEXEC 'SETMSG MSG(ISPC270)'
        End
      If config.0 = 0 Then
        Do
          exec_rc = 8
          Address ISPEXEC 'SETMSG MSG(ISPC286)'
        End
    End
  Do qq = 1 to config.0 while substr(config.qq,1,1) = '*'
  End
  If pos('ISPCFIGU CSECT',config.qq) = 0 ,               /* Ron */
   & pos('ISRCONFG CSECT',config.qq) = 0 Then
    Do
      exec_rc = 8
      Address ISPEXEC 'SETMSG MSG(ISPC287)'
      'FREE FI(INFILE)'
    End
  If exec_rc = 0 Then
    Do
      If substr(zcnvkwd,1,1) = "'" Then
        kwdname = "'"strip(zcnvkwd,'B',"'")"("zcnvkwdm")'"
      Else
        kwdname = strip(zcnvkwd,'B',"'")"("zcnvkwdm")"
      zkwd = kwdname
      Address ISPEXEC 'VPUT (ZKWD) SHARED'
      'ALLOC F(OUTFILE) REUSE SHR DA('kwdname')'
      If rc > 0 Then
        Do
          exec_rc = 8
          Address ISPEXEC 'SETMSG MSG(ISPC261)'
          'FREE FI(INFILE)'
        End
    End
return

/* Get_Default_Values: Read old and new names and defaults from ****/
/*                     The comments in this exec (See below) *******/
/*                     And place in stem variable defaults.  *******/

get_default_values: procedure expose defaultlines defaults. oldnames ,
                                     groups group. grouptitle. maxlen.
  a = 1
  defaultlines= 0
  defaults.='??';
  oldnames=''
  inlist='NO'
  do until substr(line,1,5)='**END'
     line = translate(sourceline(a))
     If substr(line,1,8)='        ' | ,        /* Shift if    OW40528*/
        datatype(substr(line,1,8),'N')=1 Then  /* blank or    OW40528*/
        line=delstr(line,1,8)                  /* line num    OW40528*/
     if inlist='NO' & substr(line,1,7)='/*START' then
       inlist='YES'
     if inlist='YES' & substr(line,1,7)='**END' then
       inlist='NO '
     if inlist = 'YES' & substr(line,1,2)¬='/*' then
       do
         parse var line oldname title
         if oldname ='*GROUP' then
           do
             groups = groups + 1
             grouptitle.groups = space(title)
           end
         else
           do
             parse var title title '=' .
             maxlen.groups=max(maxlen.groups,length(space(title))+2)
             oldnames=oldnames oldname
             defaultlines=defaultlines+1
             group.defaultlines = groups
             defaults.defaultlines=space(line)
           end
       end
     a=a+1
  end
  return

/* Get_Values_From_Asm_Source: Parse assembler source into *********/
/*                              Stem variable configname.  *********/

get_values_from_asm_source: procedure expose configname. config. ,
                            continue
  configname.='??'
  do a = 1 to config.0
    parse upper var config.a cname dc cvalue comment
    if dc = 'DC' then
      do
        type=substr(cvalue,1,1)
        select
          when(type='C') then parse var config.a "'"cvalue"' " ./*Ron*/
          when(type='F') then parse var cvalue . "'"cvalue"'" .
          when(type='A') then parse var cvalue . "("cvalue")" .
          when(type='H') then parse var cvalue . "'"cvalue"'" .
          when(type='X') then parse var cvalue . "'"cvalue"'" . /*Ron*/
          otherwise nop
        end
        cvalue=remove_double_ampersands(cvalue)
        line=cname cvalue
        configname.cname=line
      end
    if substr(config.a,72,1)¬=' ' then /* If continuation in col 72*/
      continue = 1                     /* Mark for a later message */
  end
  return

/* Remove_Double_Ampersands: Convert && To & From assembler source**/
remove_double_ampersands:
  arg setting
  do until p=0
    p=pos('&&',setting)
    if p>0 then setting=delstr(setting,p,1)
  end
  return setting

/* Add_Color: Annotate lines with 3 digit color numbers*/
add_color:
  Parse arg val
  parse var val c 2 i 3 h 4 .
  output.outlines=strip(output.outlines,'T')
  p=pos('*/',output.outlines)
  if p>0 then output.outlines=strip(delstr(output.outlines,p,2))
  output.outlines=output.outlines' /*'||colors.c intens.i hilite.h
  output.outlines=left(output.outlines,max(lengtH(line),70))||'*/'
  return

/* Addline: Generic routine to add a line to the output file  ******/
addline: procedure expose outlines output. indent pass colork. ,
                          colors. intens. hilite.
  if pass=1 then return
  parse arg line
  line = strip(line,'T')
  outlines = outlines + 1
  if substr(line,1,2)='/*' & substr(line,length(line)-1,2)¬='*/' then
   line=left(line,max(lengtH(line),70))||'*/'
  output.outlines = copies(' ',indent)line
  parse var output.outlines key . val
  ck = 1
  do while colork.ck ¬= '' & colork.ck ¬= key
    ck = ck + 1
  End
  if colork.ck = key then
    call add_color val
  return

/* View_Output: View and free data sets ****************************/
view_output:
  j = 0
  old_key = ''
  Do i = 1 to outlines
    Parse var output.i key . val
    If i > 1 & (key = old_key & key ¬= '/*') Then
      iterate
    Else
      Do
        old_key = key
        j = j + 1
        output2.j = output.i
      End
  End
  if continue=1 then
   do
    say 'A contination character was found in Column 72 of the',
        'assembler source.'
    say 'Results may not be complete.'
   end
  'EXECIO 'j' DISKW OUTFILE (FINIS STEM OUTPUT2.'
  execio_rc = rc
  If execio_rc = 0 Then
    Do
      'ALLOC REU F(OUTFILE) DA('zcnvkwd') SHR'
      member = 'MEMBER('zcnvkwdm')'
      address ispexec 'LMINIT DATAID(CNVRT) DDNAME(OUTFILE) ENQ(SHRW)'
      address ispexec 'VIEW DATAID('cnvrt')' member
      If member ¬= '' Then
        address ispexec 'LMMSTATS DATAID('cnvrt')' member
      address ispexec 'LMFREE DATAID('cnvrt')'
    End
  Else
    Address ISPEXEC 'SETMSG MSG(ISPC289)'
return
