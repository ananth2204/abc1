)ATTR DEFAULT(%+_)
 $ TYPE(OUTPUT) INTENS(LOW)  CAPS(OFF) JUST(LEFT) COLOR(TURQUOISE)
 _ TYPE(INPUT)  INTENS(HIGH)
)BODY WINDOW(45,4)
$message
+Are you sure ?_Z  +
+
+Press %ENTER+to process or press%END+to exit
)INIT
 .zvars = '(ANSWER)'
 .CURSOR = ANSWER
 &ANSWER = 'YES'
)REINIT
)PROC
 VER (&ANSWER,NB)
 VER (&ANSWER,LIST,YES,NO)
 &CKEY = .PFKEY
 IF (&CKEY = ' ')
     &CKEY = 'ENTER'
 VPUT (CKEY) SHARED
)End
