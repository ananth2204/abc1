* ******************************************************************* *
* ******************************************************************* *
*                                                                     *
* THIS SUBROUTINE SCANS A NAME FIELD PASSED TO IT BY THE CALLING      *
* RPG MAIN PROGRAM - THE FORMAT OF THE FIELD PASSED IS ASSUMED TO     *
* BE: FIRST I. LAST - AND REVERSES THE ORDER OF THE FIELD ELEMENTS    *
* TO CONSTRUCT A NAME SUITABLE FOR SORTING IN ORDER TO PLACE THE      *
* NAME INTO ALPHABETICAL SEQUENCE ON THE SURNAME (LAST) PORTION OF    *
* THE FIELD - LAST, FIRST I                                           *
*                                                                     *

* THIS IS INTENDED AS AN ILLUSTRATION OF CALLING A SUBROUTINE BY AN   *
* RPG PROGRAM                                                         *
*                                                                     *

*                                                                     *
*     .-OPERATION   .-FACTOR 2   .-RESULT   .-LENGTH                  *
*     EXIT          EX5SUB                                            *
*     RLABL                      FULLNAME   35                        *
*     ULABL                      SORTNAME   35                        *
*                                                                     *
* ******************************************************************* *
* ******************************************************************* *
*

*
         ENTRY SNAME                   MAKE SNAME VISABLE TO RPG
*
         PRINT NOGEN
         SAVE  (14,12)                 SAVE CALLER'S REGISTERS
*
         BALR  R12,0                   ESTABLISH BASE REGISTER
         USING *,R12                   ASSUME BASE REGISTER
*
         L     R10,=V(NAME)            LOAD ADDRESS OF RPG FIELD
*
         LA    R3,34(,R10)             STARTING AT END OF NAME
*
LOOP1    CLI   0(R3),C' '              LOOK FOR NON-BLANK
         BNE   CONT1                   FOUND, NEXT STEP
         BCT   R3,LOOP1                CONTINUE LOOPING
*
CONT1    LA    R4,1(,R3)               A(LAST NAME ENDS)
*
LOOP2    CLI   0(R3),C' '              LOOKING FOR BLANK
         BE    CONT2                   FOUND, NEXT STEP
         BCT   R3,LOOP2                CONTINUE LOOPING
*
CONT2    LA    R5,1(,R3)               A(LAST NAME BEGINS)
*
LOOP3    CLI   0(R3),C'.'              LOOKING FOR PERIOD
         BE    CONT3                   FOUND, NEXT STEP
         BCT   R3,LOOP3                CONTINUE LOOPING
*
CONT3    BCTR  R3,0                    MOVE BEYOND PERIOD

*
