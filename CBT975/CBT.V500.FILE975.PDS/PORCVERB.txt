      * FUNCION     : PGM THAT CALCULATES THE AMOUNT OF RESERVED WORDS *
      *               IN A COBOL ENTERPRISE 3.4.0 PGM                  *
      *               LIST ON A SYSOUT:                                *
      *               -TOTAL OF FOUND WORDS                            *
      *               -TOTAL OF RESERVED WORDS(1)                      *
      *               -TOTAL OF COMMENTS                               *
      *               -TOTAL OF LINES READ                             *
      *               -(EACH RESERVED WORD) W/ITS TOTAL AND PERCENTAGE *
      *               (1) THE RESERVED WORDS THAT ARE BETWEEN QUOTES   *
      *               DOES NOT COUNT AS IF THEY WERE. ALL THAT ARE     *
      *               BETWEEN QUOTES COUNTS AS A SPACES.               *
      *                                                                *
      *               NOTE:                                            *
      *               THE RESERVED WORDS ARE ORDER BY ALPHABET TO      *
      *               SIMPLIFY ITS CHANGE (OTHER COBOL VERSION).       *
      *               THE PROCESS OF THE PGM SHOULD TAKE LESS CPU IF   *
      *               WE ORDER BY VALUE OF THE WORDS (IF/MOVE/ETC ARE  *
      *               MORE IMPORTANT THAT SEQUENTIAL OR COMMA)         *

       2100-CUENTO-PALABRAS.
      ******************************************************************

           PERFORM VARYING WS-I FROM 1 BY 1
                     UNTIL WS-I > 80
              IF WS-ENTRADA(WS-I:1) NOT = QUOTE
                 IF WS-ENTRADA(WS-I:1) NOT = SPACES
                    IF NOT WS-WSTR
                       SET WS-WSTR     TO TRUE
                    END-IF
                 ELSE
                    IF  WS-ENTRADA(WS-I:1) = SPACES
                    AND WS-WSTR
                        SET WS-WEND    TO TRUE
                        ADD 1          TO WS-WORD
                    END-IF
                 END-IF
              END-IF
           END-PERFORM.

      ******************************************************************
       2200-SACO-LITERALES.
      ******************************************************************
      *--- CHANGE 'XXXX' FOR SPACES

           SET NO-INICIO               TO TRUE

           PERFORM VARYING WS-I FROM 1 BY 1
                     UNTIL WS-I > 80
               IF WS-ENTRADA(WS-I:1) = QUOTE
                  IF NO-INICIO
                     SET  SI-INICIO    TO TRUE
                     MOVE WS-I         TO WS-PI
                  ELSE
                     MOVE WS-I         TO WS-PF
                     ADD  1            TO WS-PF
                     SUBTRACT WS-PI  FROM WS-PF
                       GIVING WS-CANT
                     MOVE SPACES       TO WS-ENTRADA(WS-PI:WS-CANT)
                     SET  NO-INICIO    TO TRUE
                  END-IF
               END-IF
           END-PERFORM.

      ******************************************************************
       8000-READ-INPUT.
      ******************************************************************

           SET  SIN-COMENTARIO         TO TRUE
           MOVE SPACES                 TO WS-ENTRADA

           READ ENTRADA

           EVALUATE TRUE
           WHEN FS-ENTRADA-OK
                ADD 1                  TO WS-LEIDOS-ENTRADA
                MOVE REG-ENTRADA       TO WS-ENTRADA

                IF WS-ENTRADA(7:1) = '*'
                   ADD 1               TO WS-LINEAS-COMENT
                   SET CON-COMENTARIO  TO TRUE
                END-IF
           WHEN FS-ENTRADA-EOF
                CONTINUE
           WHEN OTHER
                DISPLAY 'ERROR READ INPUT'
                DISPLAY 'FILE STATUS: ' FS-ENTRADA
                PERFORM 9999-CANCELACION
           END-EVALUATE.

      ******************************************************************
       9000-FINAL.
      ******************************************************************

           CLOSE ENTRADA

           PERFORM 9100-MUEVO-EDITADAS
           PERFORM 9200-CALC-PORC
           PERFORM 9300-DISPLAY-FINAL.

      ******************************************************************
       9100-MUEVO-EDITADAS.
      ******************************************************************

           MOVE WS-TOTAL               TO WS-TOTALE
           SUBTRACT WS-LINEAS-COMENT FROM WS-LEIDOS-ENTRADA
             GIVING WS-LEIDOS-PROCE
           MOVE WS-LINEAS-COMENT       TO WS-LINEAS-COMENTE
           MOVE WS-LEIDOS-ENTRADA      TO WS-LEIDOS-ENTRADAE

           PERFORM VARYING WS-IND FROM 1 BY 1
                     UNTIL WS-IND > WS-TOPE
               MOVE WS-J(WS-IND)       TO WS-JE(WS-IND)
           END-PERFORM

           MOVE WS-WORD                TO WS-WORDE.

      ******************************************************************
       9200-CALC-PORC.
      ******************************************************************

           PERFORM VARYING WS-IND FROM 1 BY 1
                     UNTIL WS-IND > WS-TOPE
               IF WS-TOTAL > ZEROS
                  COMPUTE WS-PORC = (WS-J(WS-IND) * 100) / WS-TOTAL
                  MOVE WS-PORC         TO WS-JP(WS-IND)
                  MOVE ZEROS           TO WS-PORC
               ELSE
                  MOVE ZEROS           TO WS-JP(WS-IND)
               END-IF
           END-PERFORM.

      ******************************************************************
       9300-DISPLAY-FINAL.
      ******************************************************************

           DISPLAY 'A*******************************************'
           DISPLAY 'B***      PERCENTAGE OF USED WORDS       ***'
           DISPLAY 'C*******************************************'
           DISPLAY 'D TOTAL OF RESERVED WORDS   : ' WS-TOTALE
           DISPLAY 'E TOTAL OF FOUND WORDS      : ' WS-WORDE
           DISPLAY 'F TOTAL OF COMMENTS         : ' WS-LINEAS-COMENTE
           DISPLAY 'G TOTAL OF PROCESSED WORDS  : ' WS-LEIDOS-PROCE
           DISPLAY 'H TOTAL OF LINES READ       : ' WS-LEIDOS-ENTRADAE

           PERFORM VARYING WS-IND FROM 1 BY 1
                     UNTIL WS-IND > WS-TOPE
               DISPLAY WS-JL(WS-IND) '       : '
                       WS-JE(WS-IND) ' -' WS-JP(WS-IND)
           END-PERFORM.

      ******************************************************************
       9999-CANCELACION.
      ******************************************************************

           CLOSE ENTRADA

           DISPLAY 'PROCESSED WORDS: ' WS-LEIDOS-ENTRADA

           MOVE 12                     TO RETURN-CODE
           STOP RUN.
