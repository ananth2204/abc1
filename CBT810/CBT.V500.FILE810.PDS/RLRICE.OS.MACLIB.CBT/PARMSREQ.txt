         MACRO
         PARMSREQ &DSECT=YES
         AIF   ('&DSECT' EQ 'YES').LBL0010
PARMSREQ        DS    0A
         AGO   .LBL0020
.LBL0010 ANOP
PARMSREQ        DSECT
.LBL0020 ANOP
         MEND
