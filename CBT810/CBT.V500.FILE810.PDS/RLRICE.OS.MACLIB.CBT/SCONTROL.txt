         MACRO
         SCONTROL  &PREFIX,&SUFFIX
         LCLC      &P,&S
         AIF       (T'&PREFIX EQ 'O').NOP
&P       SETC      '&PREFIX._'
.NOP     ANOP
         AIF       (T'&SUFFIX EQ 'O').NOS
&S       SETC      '_&SUFFIX'
.NOS     ANOP
&P.DATASPACE&S                       DS 0A
&P.DATASPACE&S._ALET                 DC F'-1'   ALET OF DATA SPACE
&P.DATASPACE&S._FIRST_RECORD         DC F'1'    LOWEST RECORD SAVED
&P.DATASPACE&S._LAST_RECORD          DC F'0'    HIGHEST RECORD NBR USED
&P.DATASPACE&S._LONGEST_RECORD       DC F'0'    LONGEST RECORD
&P.DATASPACE&S._DISPLAY_FIRST_RECORD DC F'1'    FIRST RECORD TO DISPLAY
&P.DATASPACE&S._DISPLAY_LAST_RECORD  DC F'0'    LAST RECORD ON DISPLAY
&P.DATASPACE&S._DISPLAY_FIRST_COLUMN DC F'1'    FIRST COLUMN TO DISPLAY
&P.DATASPACE&S._DISPLAY_LAST_COLUMN  DC F'0'    LAST COLUMN ON DISPLAY
&P.DATASPACE&S._BYTES                DC F'0'    BYTES USED
&P.DATASPACE&S._LAST_ADDRESS         DC A(0)    LAST ADDRESS USED
&P.DATASPACE&S._BLOCKS               DC A(0)    SIZE (4K BLOCKS)
&P.DATASPACE&S._TOKEN                DC XL8'00' DATA SPACE TOKEN
&P.DATASPACE&S._NAME                 DC CL8' '  DATASPACE NAME
         MEND
