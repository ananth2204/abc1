   ADDRESS TSO ALLOC "F(OSOPTS)    DA(.......OS.OPTS)    SHR REUSE"

   ADDRESS TSO
        ISPEXEC "LIBDEF ISPPLIB DATASET ID('.......OS.ISPPLIB')"

   ADDRESS TSO
        ALLOC "FI(STEPLIBX) DA('.......OS.LOADLIB' ***) SHR REUSE"

   XX=TSOPEXT

      "OSMAIN" OSARG

   IF XX = '' THEN
      ADDRESS TSO FREE "FI(STEPLIBX)"
   ELSE
      ADDRESS TSO ALLOC "F(STEPLIBX) DA(" XX ") SHR REUSE"
