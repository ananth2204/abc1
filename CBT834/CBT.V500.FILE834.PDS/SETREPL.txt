PROC 1 NEWVALUE
/* COMMAND CHANGES VALUE OF VARIABLE ZMCREP.
/*   THIS VARIABLE CONTROLS IF TARGET DATA WILL BE
/*   REPLACED DURING COPY OR MOVE OPERATION.
/*
/*CONTROL LIST SYMLIST CONLIST MSG MAIN

 SELECT (&NEWVALUE)
   WHEN (YES) DO
     SET &ZMCREP = REPLACE
   END
   WHEN (NO) DO
     SET &ZMCREP =
   END
   OTHERWISE DO
     WRITE VARIABLE MUST BE YES OR NO.
     EXIT
   END
 END
   ISPEXEC VPUT ZMCREP PROFILE
 EXIT
