del:macro;     /* MVS Delete data set as specified by the User */

dcl %f1                    init('');
dcl %disk                  init('');

/* If User codes a parameter, then execute the DEL immediately,
   otherwise we will show the user a menu */

   if %list(1) ¬= '' then %f1='%list(1)';

   if %f1 ¬= '' then goto gotfile;

loop:

Panel  box
/( ctr hi 'Jol DELETE Input Panel')
///
/('  You can Type the Name of the Data Sets You Wish to Delete,')
 ('  OR You can Select the Data Set from the Jol Data Set Data Base.')
////
/
('         Disk          ====> ',disk  ,6,'%disk')
/
('         Delete File   ====> ',f1,44,'%f1')
////

helpline
('Enter Starts Processing, F3 Exits, F4 shows Files.');

checkf1:
     if Syspfk=3 then exit;

     if %syspfk=4
     then do;
         dbfshow  all 'File to Delete';
         f1='%showsel';
     end;
     if f1=''
     then do;
         panel reread from f1 'You must supply a file name';
         goto checkf1;
     end;

   if syspfk=3 then exit;   /* Back to Caller */

gotfile:
   if %disk ¬='' then %disk=' From %disk';
   if %BUILDJOB='YES'
   then do;
       BLD04P                  '   Delete %f1 %disk;';
       exit;
   end;
   delete %f1 %disk;


end;
