FULLQUAL: MACRO;
IF %#####1 THEN;
           ELSE DO; DCL %#####1 INIT '' EXT;
                    DCL %#####2 INIT '' EXT;
                    DCL %#####3 INIT '' EXT;
                    END;
%#####1 = %LIST(1);
%#####2 = %LIST(2);
%#####3 = '';
IF %LIST(3) THEN %#####3 = %LIST(3);
END;
%####TP = TYPE(%#####1);
%####L = LENGTH(%#####1);
%####FN = SUBSTR(%#####1,1,%####L);
IF %####TP = 'LIT'
  THEN %###FQN = '%####FN';
  ELSE DO; %####I = INDEX(####FN,'(');
           IF %####I = 0 THEN %####LL = %####L;
                         ELSE %####LL = %####I - 1;
           %####VQ = 0;
           %####L2 = LENGTH(#####2);
           %####LQS = %####LL - %####L2;
           %####LQS = %####LQS + 1;
           IF %####LQS > 0
             THEN DO; %####LAST = SUBSTR(####FN,%####LQS,%####L2);
                      IF '%####LAST' = '%#####2' THEN %####VQ = 1;
                      END;
           IF %#####3 Â¬= ''
             THEN DO; %####L3 = LENGTH(#####3);
                      %####LQS = %####LL - %####L3;
                      %####LQS = %####LQS + 1;
                      IF %####LQS > 0
                      THEN
                      DO; %####LAST = SUBSTR(####FN,%####LQS,%####L3);
                          IF '%####LAST' = '%#####3' THEN %####VQ = 1;
                          END;
                      IF %####VQ = 0
                        THEN DO; %MESS = 'VALID QUALIFIERS FOR %####FN '
                                   || 'ARE %#####2 AND %#####3. ';
                                 %MESS = '%MESS ENTER QUALIFIER.';
                                 WRITE '%MESS';
                                 READ #####2;
                                 END;
                      END;
           IF %####I = 0
             THEN DO; SET ###FQN = '%SYSPREF.%####FN';
                      IF %####VQ = 0 THEN %###FQN = '%###FQN.%#####2';
                      END;
             ELSE DO; %####I1 = %####I - 1;
                      %####S1 = SUBSTR(####FN,1,%####I1);
                      %####S2 = SUBSTR(####FN,%####I);
                      IF %####S1 = '' THEN %####QUAL = %#####2;
                                      ELSE %####QUAL = '.%#####2';
                      IF %####VQ = 1 THEN %####QUAL = '';
                      %###FQN = '%SYSPREF.%####S1%####QUAL%####S2';
                      END;
           END;
