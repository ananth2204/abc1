 LISTSRC: MACRO (%PRJCAT=,%GEMLIB=,%MAC1=,%NOPANEL,
     %PASSWD=,%SYSOUT=,%GEMMOD,%OWNER=);
 %OWNER = OWNERID;
 IF %SYSOUT = '' THEN %SYSOUT= ' *';
 IF %PRJCAT= '' THEN DO;
    %PRJCAT = 'GEM.SOURCE.CATALOG';  /* DEFAULT FOR AIR NZ   */
 END;
 IF %GEMLIB='' THEN DO;
    %GEMLIB = 'GEN';  /* DEFAULT FOR AIR NZ  */
 END;
 IF %LIST(1) ¬= '' THEN %GEMMOD = '%LIST(1)';
 IF %PASSWD ¬= '' THEN %PASSWD = '/(%PASSWD)';
 IF %NOPANEL = '' THEN
 PANEL ('          LIST SOURCE OPTIONS   ')
 //('            ANY HIGHLIGHTED FIELD CAN BE CHANGED')
   (' NAME OF GEM MODULE =================>',GEMMOD,12,'%GEMMOD')
   (' SYSOUT CLASS =======================>',SYSOUT,1,'%SYSOUT')
   (' GEM PROJECT CATALOGUE ==============>',PRJCAT,44,'%PRJCAT')
   (' SOURCE GEM LIBRARY =================>',GEMLIB,3,'%GEMLIB')
 ///('        *****************************************************');
 DCL PRNTOUT DS DSN &&PRNT(%GEMMOD) UNIT SYSDA
      2,2,2 CYLS FB 80,19040;
 DCL GEMPRT SYSOUT %SYSOUT;
 DCL PUTPARM * REPLACE;
 - LIBRARY GEM.%GEMLIB.SOURCE
 - PUT %GEMMOD,REP,PS,OUT=PRNTOUT
 EOF;
 PUT: DCL JQHGEM3 PROG NODYNAMS=1
 PRNTOUT WRITES PRNTOUT
 GEMPRJ READS %PRJCAT
 SYSIN READS PUTPARM
 SYSLIST WRITES GEMPRT
 SYSPRINT WRITES GEMPRT;
 RUN PUT;
 IF PUT >= 8 THEN DO;
 STOP 'PUT OF %GEMMOD FAILED';
 END;
 DISPLAY 'PUT OF %GEMMOD TO &&PRNT(%GEMMOD) COMPLETE';
 PRINT PRNTOUT CLASS=A;
 END;
