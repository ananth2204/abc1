   MAIN : MACRO(%SYSTEM=,%TYPE=ANY,%LINES=,%CARDS=,%HOLD=,%SETUP=,
          %CLASS=,%FAILURE=,%JOBSTEP=,%NJPCLASS=,%HOTJOB=,%JPRTY=,
          %ACMAIN=,%ACHOLD=,%IORATE=,%ORG=,%DEADLINE=,%FETCH=,
          %WARNING,%CANCEL,%DUMP);
 /*IF %SPOOL ¬= 'ASP' THEN SIGNAL
      2,'//*MAIN CARD GENERATED FOR %SPOOL' ; */
   DCL %OPTION  INIT 'TYPE=%TYPE';
   DCL %CANCTYP INIT '';
   IF %WARNING ¬='' THEN %CANCTYP=',W';
   IF %CANCEL  ¬='' THEN %CANCTYP=',C';
   IF %DUMP    ¬='' THEN %CANCTYP=',D';
   IF %CARDS    ¬= '' THEN %OPTION = '%OPTION,CARDS=%CARDS%CANCTYP' ;
   IF %LINES    ¬= '' THEN %OPTION = '%OPTION,LINES=%LINES%CANCTYP' ;
   IF %CLASS    ¬= '' THEN %OPTION = '%OPTION,CLASS=%CLASS%CANCTYP' ;
   IF %SYSTEM   ¬= '' THEN %OPTION = '%OPTION,SYSTEM=%SYSTEM' ;
   IF %SETUP    ¬= '' THEN %OPTION = '%OPTION,SETUP=%SETUP' ;
   IF %HOLD     ¬= '' THEN %OPTION = '%OPTION,HOLD=%HOLD' ;
   IF %FAILURE  ¬= '' THEN %OPTION = '%OPTION,FAILURE=%FAILURE' ;
   IF %JOBSTEP  ¬= '' THEN %OPTION = '%OPTION,JOBSTEP=%JOBSTEP' ;
   IF %NJPCLASS ¬= '' THEN %OPTION = '%OPTION,NJPCLASS=%NJPCLASS' ;
   IF %HOTJOB   ¬= '' THEN %OPTION = '%OPTION,HOTJOB=%HOTJOB' ;
   IF %ACMAIN   ¬= '' THEN %OPTION = '%OPTION,ACMAIN=%ACMAIN' ;
   IF %ACHOLD   ¬= '' THEN %OPTION = '%OPTION,ACHOLD=%ACHOLD' ;
   IF %IORATE   ¬= '' THEN %OPTION = '%OPTION,IORATE=%IORATE' ;
   IF %ORG      ¬= '' THEN %OPTION = '%OPTION,ORG=%ORG' ;
   IF %DEADLINE ¬= '' THEN %OPTION = '%OPTION,DEADLINE=%DEADLINE' ;
   IF %FETCH    ¬= '' THEN %OPTION = '%OPTION,FETCH=%FETCH' ;
   IF %JPRTY    ¬= '' THEN %OPTION = '%OPTION,JPRTY=%JPRTY' ;
   OPCNTL '//*MAIN %OPTION' AFTER JOBCARD ;
   END ;
