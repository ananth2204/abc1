type:macro;     /* MVS List/TYPE data set as specified by the User */


dcl %f1                    init('');
dcl %disk                  init('');

/* If User codes a parameter, then execute the TYPE immediately,
   otherwise we will show the user a menu */

   if %list(1) ¬= '' then %f1='%list(1)';

   if %f1 ¬= '' then goto gotfile;

loop:

Panel  box
/( ctr hi 'Jol TYPE Input Panel')
///
/('  You can Type the Name of the Data Sets You Wish to Type,')
 ('  OR You can Select the Data Set from the Jol Data Set Data Base.')
////
/
('         Disk          ====> ',disk  ,6,'%disk')
/
('         Type File     ====> ',f1,44,'%f1')
////
(ctr 'The next Panels allow Other Files to be Selected')

helpline
('Enter Starts Processing, F3 Exits, F4 shows Files.');

checkf1:
     if Syspfk=3 then exit;

     if %syspfk=4
     then do;
         dbfshow  all 'File to Type';
         f1='%showsel';
     end;
     if f1=''
     then do;
         panel reread from f1 'You must supply a file name';
         goto checkf1;
     end;

   if syspfk=3 then exit;   /* Back to Caller */

gotfile:
   if %disk ¬='' then %disk=' Vol %disk';
   if %BUILDJOB='YES'
   then do;
       BLD04P                  '   Print %f1 %disk;';
       exit;
   end;
   Print  %f1 %disk;


end;
