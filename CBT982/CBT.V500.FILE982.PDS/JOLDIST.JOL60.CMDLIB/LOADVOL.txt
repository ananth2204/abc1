 JCLTOJOL: MACRO(%LIB='TEST.PROCLIB');
     PANEL ('JCL TO JOL CONVERTOR COMMAND')
    //   ('INPUT CATALOGUED PROCEDURE LIBRARY =====>',LIB,44,'%LIB')
    //   ('MEMBER                             =====>',MEM,8,'%LIST(1)')
    //   ('DATA SET WHERE JOL WILL BE PLACED  =====>',OUT,44,
          '%SYSUID.JOL');
 CONVERT : DCL JCLTOJOL PROG LIB JOL33.LOAD
           JCLIN READS %LIB(%MEM)
           JOLCODE UPDATES %OUT (%MEM)
           JOLLIST WRITES PRINTER
           SYSPRINT WRITES PRINTER
           JCLOUT WRITES PRINTER
           PLIDUMP WRITES PRINTER;
  DCL PRINTER PRINTER NODCB;
 RUN CONVERT 'ISA(30K)/';
  END;
