* MEMBER J40XACAT
* THIS PTF CORRECTS A PROBLEM WITH XA CATALOG LOOKUPS.
*
* * - MVS/XA WILL NOT ALLOW GDGS OF 4 DIGITS, EG A.B(0001) IS ILLEGAL.
*
******************************************************************
*
* FORCE OUTPUT OF $$VREF DD CARDS WHEN NOCAT IS SPECIFIED.
*
 NAME UJV00MN UJS01CAT
 BASE 0D90
 VER 0EC4 D204,1001,3049 MOVE GDG NUMBER IN
 VER 0ECA 925D,1006      MOVE ')' NUMBER IN
 VER 1A24 BC94,BC96      CHECK ZAP AREA
 REP 1A24 D200,1001,3049 MOVE '(' IN TO FIELD.
 REP 1A2A D202,1002,304B MOVE GDG NUMBER IN
 REP 1A30 925D,1005      MOVE ')' TO FIELD
 REP 1A34 47F0,B13E      BACK TO MAINLINE
 REP 0EC4 47F0,BC94      TURN ZAP ON BY BRANCHING TO ZAP AREA
 IDRDATA  MVSXACAT
