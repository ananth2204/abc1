 PARM GRAPHICS(CHAIN(SN))
 /*                          */
 DELETE   &DBPFX..DSN    CLUSTER
 DELETE   &DBPFX..MBR    CLUSTER
 SET MAXCC = 0
 /* Dataset database         */
 DEFINE CLUSTER( -
          NAME(&DBPFX..DSN)     -
          CYLINDER(2 2) -
          KEYS(12 0)  -
          RECORDSIZE(50 100) -
          SHR(3,3) -
          VOL(&DBVOL) -
         ) -
         INDEX(CISZ(4096))
 /* Member database         */
 DEFINE CLUSTER( -
          NAME(&DBPFX..MBR)     -
          CYLINDER(20 20) -
          KEYS(20 0)  -
          RECORDSIZE(50 100) -
          SHR(3,3) -
          VOL(&DBVOL) -
         ) -
         INDEX(CISZ(4096))
 REPRO INFILE(MBRI) OUTDATASET(&DBPFX..MBR)
