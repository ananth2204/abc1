//CMDSHR   PROC  PRM='WARM',SYS='MVSR'
//*
//*   NAME:  CMDSHR
//*   PURPOSE:  TO SHARE COMMANDS FROM ONE SYSTEM TO ANOTHER
//*   NOTE:     PARM MAY BE EITHER COLD OR WARM.
//*             'COLD' MUST BE SPECIFIED ON ONE SYSTEM AND ONLY ON THE
//*             FIRST INVOKATION.  'WARM' IS USED THERE AFTER ON ALL
//*             SYSTEMS.
//*             SYS= MUST BE SET TO THE SYSTEM YOU ARE RUNNING ON
//*
//*
//WAITTING EXEC  PGM=CMDSHR,
//             PERFORM=9,
//             PARM='&PRM'
//STEPLIB   DD  DISP=SHR,DSN=SYSS.LINKLIB
//CSLOG     DD  DISP=SHR,DSN=SYSS.CMDSHR.&SYS..LOG
//SNAPDUMP  DD  DISP=SHR,DSN=SYSS.CMDSHR.&SYS..SNAPDUMP
//SYSUDUMP  DD  DISP=SHR,DSN=SYSS.CMDSHR.&SYS..DUMP
//CSDS      DD  DISP=SHR,DSN=SYSS.CMDSHR.DS
//CSIN      DD  DISP=SHR,DSN=SYS1.PARMLIB(CMDSHR)
