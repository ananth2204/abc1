/* We first need rexx in comment if compiled or read from SYSPROC. */
pgm_REXR0465:
  rexxpgm_version = "01.00"          /* Set with 'ver 1' and 'lev 2'. */

  SAY ""
  SAY "REXR0465: X2D"
  SAY "============="

  bit_str = "11110001"
  hex_val = B2X(bit_str)
  bin_val = X2B(hex_val)
  dec_val = X2D(hex_val)
  Say "BIT("bit_str") is",
     "Hex("hex_val") is",
     "Bin("bin_val") is",
     "Dec("dec_val")."

  RETURN 0
