rexx_Main_IRRRXALL:
  return_code = rexx_Initialize()
  If (return_code = 0) then,
    Do
      return_code = rexx_Process()
      return_code = rexx_Termination()
    End
  Return 0

/* >>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>
=======================================================================
 In this routine we open any files and initialize any
 constants we might need.
=======================================================================
<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<< */
rexx_Initialize:
  PARSE VERSION,
    version_rexx_name,
    version_rexx_version,
    version_rexx_date
  PARSE SOURCE,
    source_host_env,
    source_call_type,
    source_uc_exec_name,
    source_hostdd,
    source_hostdsn,
    source_nf_exec_name,
    source_initial_cmd_env,
    source_asid_name,
    source_char8_user_token,
    source_hostother
  job_specs = rexx_jobspecs()
  sysvar_sysuid = SYSVAR("SYSUID")
  sysvar_sysenv = SYSVAR("SYSENV")
  say ""
  say Copies("=",55)
  say left("=" job_specs,53) "="
  say left("=" source_uc_exec_name,
    "executing under" source_host_env,
    "as a" source_call_type,53) "="
  say left("= on" date() "at" time()".",53) "="
  say Copies("=",55)
  say ""
  return_code = 0

  If (source_host_env >< "TSO") Then do
    Say "Wrong host environment" source_host_env
    return_code = 16
  End

  CVT    = C2d(Storage(10,4))                /* point to CVT         */
  CVTRAC = C2d(Storage(D2x(CVT + 992),4))    /* point to RACF CVT    */
  RCVTID = Storage(D2x(CVTRAC),4)            /* point to RCVTID      */
  RACFVRM = Storage(D2x(CVTRAC + 616),4)     /* RACF Ver/Rel/Mod     */
  select
    when(RCVTID = "RCVT") then,
      Do
        SECNAM = "RACF"
        say "Okay, you are running RACF. I mean Security Server."
        say "VRM("racfvrm")"
      End
    when(RCVTID = "RTSS") then,
      Do
        SECNAM = "Top Secret"
        say "You are running Top Secret," source_uc_exec_name,
          "is a RACF parser."
        say "VRM("racfvrm")"
        return 12
      End
    when(RCVTID = "ACF2") then,
      Do
        SECNAM = "CA-ACF2"
        say "You are running CA-ACF2," source_uc_exec_name,
          "is a RACF parser."
        say "VRM("racfvrm")"
        return 12
      End
    Otherwise,
      SECNAM = RCVTID
      say "You are running something other than RACF,",
        source_uc_exec_name "is a RACF parser."
      say "VRM("racfvrm")"
      return 12
  End

  mvsvar_sysname = MVSVAR("SYSNAME")

  irrdbu00_ld. = ""

  allcblk = ddbycblk()
  call de_parse
  find_dd = "IRRDBU00"
  found_dd = tiotddnm.find_dd
  if (found_dd = "") then,
    do
      Say "Attempting to allocate" find_dd"."
      "ALLOC FILE("find_dd")",
        "DATASET(RACF."mvsvar_sysname") SHR REUSE"
      dmy1 = listdsi(find_dd "FILE")     /* get data set info for dd */
      if dmy1 = 0 then
        inname1 = SYSDSNAME     /* copy dsname if LISTDSI successful */
      else
        inname1 = "????????"     /* else use ????????                */
      say "Reading DDname("find_dd")"
      say "  DSN="inname1
    end
  else,
    do
      say "Reading DDname("find_dd")"
      found_dsns = dd_dsname.found_dd.0
      do ds_x = 1 to found_dsns
        say "  DSN="dd_dsname.found_dd.ds_x
      end
    end
  trap_msg = OutTrap("io_msgs.")
  quietly = Msg("OFF")
  Address MVS "EXECIO",
    "* DISKR" find_dd "0 (FINIS STEM RACFDATA.)"
  read_rc = rc
  trap_msg = OutTrap("OFF")
  noisily = Msg("ON")
  if (read_rc >< 0) then,
    do
      say "Read error on" find_dd ", RC("read_rc")"
      if (racfdata.0 = 0) then,
        return read_rc
    end
  say "Records read ("racfdata.0") from" find_dd"."
  call load_descriptors
  pp = ""
  ml_pp = 0
  id = ""
  ml_id = 0
  Return return_code

/* >>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>
=======================================================================
 In this routine we do our processing.
=======================================================================
<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<< */
rexx_Process:
  do ix = racfdata.0 to 1 by -1
    curr_id = left(racfdata.ix,4)
    call count_id
    parse var curr_id id_pp +2 id_ss +1 id_ff +1
    call count_pp
    select
      when (id_pp = "01") then call parse_01_group
      when (id_pp = "02") then call parse_02_user
      when (id_pp = "04") then call parse_04_dataset
      when (id_pp = "05") then call parse_05_resource
      otherwise
        say "I don't recognize ("id_pp")."
    end
    if (any_extra >< "") then,
      do
        say "Error: type("curr_id")",
          "Expected length("irrdbu00_maxlen.curr_id")",
          "Actual length("length(racfdata.ix)")",
          "Extra Data("any_extra")"
      end
  end
  call show_pp
  call show_id
  Return return_code

<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<< */
rexx_Termination:
  Return return_code

rexx_jobspecs:
  /*---------------------------------------------------------------*/
  /* ADDRESSABILITY TO CVT FROM PSA                                */
  /*---------------------------------------------------------------*/
  PTR2PSA = 0
  CVTPTR = STORAGE(D2X(PTR2PSA + 16),4)
  /*---------------------------------------------------------------*/
  /* ADDRESSABILITY TO SMCA, THEN GET SMF SYSTEM ID                */
  /*---------------------------------------------------------------*/
  CVTSMCA = STORAGE(D2X(C2D(CVTPTR)+197),3)
  SMFID = STORAGE(D2X(C2D(CVTSMCA)+16),4)
  /*---------------------------------------------------------------*/
  /* GET JOBNAME, STEP, AND PROCEDURE                              */
  /*---------------------------------------------------------------*/
  PSATOLD  = STORAGE(21C,4)                    /* POINTER TO TCB   */
  PTRTIOT  = STORAGE(D2X(C2D(PSATOLD)+12),4)   /* POINTER TO TIOT  */
  JOBNAME  = STORAGE(D2X(C2D(PTRTIOT)),8)
  STEPNAME = STORAGE(D2X(C2D(PTRTIOT)+8),8)
  PROCNAME = STORAGE(D2X(C2D(PTRTIOT)+16),8)
  Tcbjscbb = Storage(D2x(C2d(Psatold)+181),3)              /* jscb */
  jscbssib = STORAGE(D2X(C2D(tcbjscbb)+316),4)      /* Ptr to SSIB */
  ssibjbid = STORAGE(D2X(C2D(jscbssib)+12),8)    /* job identifier */
  Return JOBNAME SSIBJBID STEPNAME PROCNAME

/* >>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>
=======================================================================
 Count how many of the major record types we processed.
=======================================================================
<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<< */
count_pp:
  if (pos(id_pp,pp) = 0) then,
    do
      ctr = 1
      pp = pp id_pp
      ctr_pp.id_pp = ctr
    end
  else
    do
      ctr = ctr_pp.id_pp + 1
      ctr_pp.id_pp = ctr
    end
  if (length(ctr) > ml_pp) then ml_pp = length(ctr)
  return

/* >>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>
=======================================================================
 Count how many of the minor record types we processed.
=======================================================================
<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<< */
count_id:
  if (pos(curr_id,id) = 0) then,
    do
      ctr = 1
      id = id curr_id
      ctr_id.curr_id = ctr
    end
  else
    do
      ctr = ctr_id.curr_id + 1
      ctr_id.curr_id = ctr
    end
  if (length(ctr) > ml_id) then ml_id = length(ctr)
  return

/* >>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>
=======================================================================
 Show how many of the major record types we processed.
=======================================================================
<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<< */
show_pp:
  rl = ml_pp
  bl = copies(" ",rl)
  do ix = 1 to words(pp)
    currpp = word(pp,ix)
    right_ctr = right(bl||ctr_pp.currpp,rl)
    say currpp "has" right_ctr "entries:" irrdbu00_description.currpp
  end
  return

/* >>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>
=======================================================================
 Show how many of the minor record types we processed.
=======================================================================
<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<< */
show_id:
  rl = ml_id
  bl = copies(" ",rl)
  do ix = 1 to words(id)
    currid = word(id,ix)
    right_ctr = right(bl||ctr_id.currid,rl)
    say left(currid,4) "has" right_ctr "entries:" ,
      irrdbu00_description.currid
  end
  return

/* >>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>
=======================================================================
 Deparse the stream off ddname information
=======================================================================
<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<< */
/* #dds;dd#;ddname;#dsn;dsn#;dsname;...                        */
de_parse:
  allofit = allcblk
  tiotddnm. = ""
  tiotddnm.0 = 0
  parse var allofit nbr_dds ";" allofit
  do dd_x = 1 to nbr_dds
    parse var allofit dd_nbr ";" dd_name ";" nbr_dsns ";" allofit
    tiotddnm.dd_name = dd_x
    dd_dsname.dd_x. = ""
    do ds_x = 1 to nbr_dsns
      parse var allofit ds_nbr ";" ds_name ";" allofit
      dd_dsname.dd_x.ds_x = ds_name
    end
    dd_dsname.dd_x.0 = ds_x - 1
  end
  tiotddnm.0 = dd_x - 1
  return
