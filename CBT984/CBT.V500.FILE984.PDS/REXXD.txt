/* rexx */
pgm_rexxd:
  PARSE ARG hlq
  IF ((hlq = "hlq") | (hlq = "")) THEN DO
    SAY "You must provide the high-level-qualifier to the product."
    RETURN 16
  END
  ADDRESS ISPEXEC "CONTROL ERRORS RETURN"
  ADDRESS ISPEXEC "LIBDEF ISPLLIB DATASET ID('"hlq".SFANLMD') STACK"
  ADDRESS ISPEXEC "LIBDEF ISPMLIB DATASET ID('"hlq".SFANMSEU') STACK"
  ADDRESS ISPEXEC "LIBDEF ISPPLIB DATASET ID('"hlq".SFANPENU') STACK"
  ADDRESS ISPEXEC "LIBDEF ISPSLIB DATASET ID('"hlq".SFANSKL') STACK"
  Address TSO "ALTLIB Activate Application(EXEC)" ,
    "DATASET('"hlq".SFANCMD') UNCOND"
  ADDRESS ISPEXEC "SELECT" ,
    "PGM(ISRFPR) PARM((FANFP14,ISRFP14) 14) NEWPOOL"
  ADDRESS TSO "ALTLIB DeActivate Application(EXEC)"
  ADDRESS ISPEXEC "LIBDEF ISPLLIB"
  ADDRESS ISPEXEC "LIBDEF ISPMLIB"
  ADDRESS ISPEXEC "LIBDEF ISPPLIB"
  ADDRESS ISPEXEC "LIBDEF ISPSLIB"
  RETURN 0
