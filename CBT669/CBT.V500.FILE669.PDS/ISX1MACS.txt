         Macro
         Terminate
         Freemain R,lv=$wrkl,a=(10)
         #module exit,rc=8
         Mend

         Macro
         XWTO  &l=,&force=N
         aif   (k'&syslist(1) eq 0).l
         mvlit wtot,&syslist(1),lr=r15
         ago   .l2
.l       aif   (k'&l eq 0).g
         $la   r15,&l
.l2      la    r15,4(,r15)
         sll   r15,16
         st    r15,wtox
.g       la    r1,wtox
         sr    r0,r0
         aif   ('&force' ne 'Y').b
         svc   35
.b       clear wtot
         Mend
