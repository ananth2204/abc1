/* REXX    */
/* TRACE R */
/* CBR3570I -- GENERAL MESSAGE FROM OAM                   */
/* THIS PROGRAM RUNS WHEN A VTS IMPORT OR EXPORT STARTS   */
ARG MESSAGE


SELECT

  WHEN POS(' E0016 ',MESSAGE) > 0 THEN
   CALL E0016

  WHEN POS(' E0017 ',MESSAGE) > 0 THEN
   CALL E0017


  OTHERWISE
    NOP  /* SHOULD NEVER GET HERE */

  END


EXIT
