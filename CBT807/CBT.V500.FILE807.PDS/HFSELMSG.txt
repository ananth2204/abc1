          ISPEXEC LIBDEF ISPPLIB DATASET ID('HRA.HFSELECT.ISPPLIB')
          ISPEXEC LIBDEF ISPMLIB DATASET ID('HRA.HFSELECT.ISPMLIB')
        SELMAIN:  ISPEXEC DISPLAY PANEL(SELMAIN)
                  IF LASTCC EQ 8 THEN EXIT
                  ISPEXEC VGET (PANELID) SHARED
                  IF &PANELID EQ SEL004 THEN GOTO SEL004
                  IF &PANELID EQ SEL021 THEN GOTO SEL021
                  IF &PANELID EQ SEL040 THEN GOTO SEL040
                  IF &PANELID EQ SEL076 THEN GOTO SEL076
                  ISPEXEC DISPLAY PANEL(&PANELID)
                  GOTO SELMAIN
        SEL004:   ISPEXEC DISPLAY PANEL(SEL004)
                  IF LASTCC EQ 8 THEN GOTO SELMAIN
                  ISPEXEC DISPLAY PANEL(SEL004B)
                  IF LASTCC EQ 8 THEN GOTO SELMAIN
                  IF &ZCMD EQ LAST THEN GOTO SEL004
                  GOTO SELMAIN
        SEL040:   ISPEXEC DISPLAY PANEL(SEL040)
                  IF LASTCC EQ 8 THEN GOTO SELMAIN
                  ISPEXEC DISPLAY PANEL(SEL040B)
                  IF LASTCC EQ 8 THEN GOTO SELMAIN
                  IF &ZCMD EQ LAST THEN GOTO SEL040
                  GOTO SELMAIN
        SEL076:   ISPEXEC DISPLAY PANEL(SEL076)
                  IF LASTCC EQ 8 THEN GOTO SELMAIN
                  ISPEXEC DISPLAY PANEL(SEL076B)
                  IF LASTCC EQ 8 THEN GOTO SELMAIN
                  IF &ZCMD EQ LAST THEN GOTO SEL076
                  GOTO SELMAIN
        SEL021:   ISPEXEC DISPLAY PANEL(SEL021)
                  IF LASTCC EQ 8 THEN GOTO SELMAIN
        SEL021B: ISPEXEC DISPLAY PANEL(SEL021B)
                  IF LASTCC EQ 8 THEN GOTO SELMAIN
                  IF &ZCMD EQ LAST THEN GOTO SEL021
        SEL021C: ISPEXEC DISPLAY PANEL(SEL021C)
                  IF LASTCC EQ 8 THEN GOTO SELMAIN
                  IF &ZCMD EQ LAST THEN GOTO SEL021B
                  ISPEXEC DISPLAY PANEL(SEL021D)
                  IF LASTCC EQ 8 THEN GOTO SELMAIN
                  IF &ZCMD EQ LAST THEN GOTO SEL021C
                  GOTO SELMAIN
                  END
