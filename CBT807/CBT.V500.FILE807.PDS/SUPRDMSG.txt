          ISPEXEC LIBDEF ISPPLIB DATASET ID('HRA.SUPRDUMP.ISPPLIB')
          ISPEXEC LIBDEF ISPMLIB DATASET ID('HRA.HFSELECT.ISPMLIB')
        DMPMAIN:  ISPEXEC DISPLAY PANEL(DMPMAIN)
                  IF LASTCC EQ 8 THEN EXIT
                  ISPEXEC VGET (PANELID) SHARED
                  IF &PANELID EQ DMP040 THEN GOTO DMP040
                  ISPEXEC DISPLAY PANEL(&PANELID)
                  GOTO DMPMAIN
        DMP040:   ISPEXEC DISPLAY PANEL(DMP040)
                  IF LASTCC EQ 8 THEN GOTO DMPMAIN
                  ISPEXEC DISPLAY PANEL(DMP040B)
                  IF LASTCC EQ 8 THEN GOTO DMPMAIN
                  IF &ZCMD EQ LAST THEN GOTO DMP040
                  GOTO DMPMAIN
                  END
