//ALGOL68C JOB (,),'UNDUMP FILES'
//JOBLIB DD DSN=A68LIB.MOD,DISP=SHR assumed to contain pack/unpack pgm
//*
//* 1.  Make the job card suitable for your installation.
//*
//* 2.  This job copies all the material from the tape to disc data
//*     sets with ds names having AL68 as the first qualifier.
//*     It does not construct an ALGOL68C system - this is done by the
//*     other undump job on the tape.
//*
//* 3.  This job assumes that the pack/unpack program is available
//*     from A68LIB.MOD (JOBLIB DD card).
//*
//* 4.  The block sizes and space allocations are suitable for 3330
//*     discs.
//*
//* 5.  The dispositions for the data sets may need changing from
//*     NEW,KEEP to NEW,CATLG depending on the installation's
//*     requirements.
//*
//*
//*
//* COPY FB80 DATA SET
//*
//F PROC N=,                  DATA SET NAME
//       S='1,1',             SPACE ALLOCATION
//       L=,                  TAPE FILE NUMBER
//       SER=A68C00,UNIT=TAPE9, TAPE SERIAL AND UNIT NAME
//       DEN=3,LAB=SL, TAPE DENSITY AND LABEL TYPE
//       BLKSIZE=3200,       TAPE BLOCKSIZE
//       TRAN=''        7TRACK RECORDING TECHNIQUE
// EXEC PGM=IEBGENER,REGION=80K
//SYSPRINT DD SYSOUT=A
//SYSIN DD DUMMY
//SYSUT1 DD UNIT=&UNIT,VOL=(,RETAIN,SER=&SER),
//          LABEL=(&L,&LAB),DISP=OLD,DSN=A68C&L,
//          DCB=(LRECL=80,BLKSIZE=&BLKSIZE,RECFM=FB,DEN=&DEN&TRAN)
//SYSUT2 DD UNIT=SYSDA,DISP=(NEW,KEEP),DSN=&N,
//          SPACE=(TRK,(&S)),DCB=(RECFM=FB,LRECL=80,BLKSIZE=2480)
// PEND
//*
//* UNPACK TO PDS
//*
//P PROC N=,                  DISC DSNAME
//       S='1,1',             SPACE ALLOCATION
//       Q=,                  'QUOTE' FOR PREFIX QUOTE STROPPING ****** ********
//       L=,                  TAPE FILE NUMBER
//       SER=A68C00,UNIT=TAPE9, TAPE SERIAL AND UNIT NAME
//       DEN=3,LAB=SL, TAPE DENSITY AND LABEL TYPE
//       BLKSIZE=3200,       TAPE BLOCKSIZE
//       TRAN=''        7TRACK RECORDING TECHNIQUE
// EXEC PGM=PACK,REGION=120K,PARM='/UNPACK,&Q'
//SYSPRINT DD SYSOUT=A
//FROM DD UNIT=&UNIT,VOL=(,RETAIN,SER=&SER),
//         LABEL=(&L,&LAB),DISP=SHR,DSN=A68C&L,
//         DCB=(LRECL=80,BLKSIZE=&BLKSIZE,RECFM=FB&TRAN,DEN=&DEN)
//UNPACKED DD UNIT=SYSDA,DISP=(NEW,KEEP),DSN=&N,
//            SPACE=(TRK,(&S))
// PEND
//*
//* UPDTE TO PDS
//*
//C PROC N=,                  DISC DSNAME
//       S='1,1',             SPACE ALLOCATION
//       L=,                  TAPE FILE NUMBER
//       SER=A68C00,UNIT=TAPE9, TAPE SERIAL AND UNIT NAME
//       DEN=3,LAB=SL, TAPE DENSITY AND LABEL TYPE
//       BLKSIZE=3200,       TAPE BLOCKSIZE
//       TRAN=''        7TRACK RECORDING TECHNIQUE
// EXEC PGM=IEBUPDTE,REGION=40K,PARM=NEW
//SYSPRINT DD DUMMY
//SYSIN DD UNIT=&UNIT,VOL=(,RETAIN,SER=&SER),
//         LABEL=(&L,&LAB),DISP=SHR,DSN=A68C&L,
//         DCB=(LRECL=80,BLKSIZE=&BLKSIZE,RECFM=FB&TRAN,DEN=&DEN)
//SYSUT2 DD UNIT=SYSDA,DISP=(NEW,KEEP),DSN=&N,
//          SPACE=(TRK,(&S)),DCB=(RECFM=FB,LRECL=80,BLKSIZE=2480)
// PEND
//*
//CHARSET EXEC F,L=1,N='AL68.CHARSET',S='1,1' CHARACTER SET
//DISTRJOB EXEC F,L=3,N='AL68.TAPEJOB',S='1,1' JOB THAT WROTE TAPE
//INST360 EXEC F,L=4,N='AL68.INST360',S='1,1'
//UNDUMP EXEC F,L=5,N='AL68.DISTRJOB.UA68C00',S='4,1' CONSTRUCT JOB
//UNDUMP1 EXEC F,L=6,N='AL68.DISTRJOB.VA68C00',S='4,1' UNDUMP FILES
//MISCOBJ EXEC C,L=7,N='AL68.LIBOBJ',S='4,4,4' LIB & XREF OBJECT
//PACKOBJ EXEC C,L=8,N='AL68.PACKOBJ',S='1,1,1' PACK/UNPACK OBJECT
//A68COBJ EXEC C,L=9,N='AL68.A68C.OBJ247',S='32,8,8' COMPILER OBJ
//Z370OBJ EXEC C,L=10,N='AL68.Z370.OBJ119',S='8,4,2' Z370 OBJECT
//PROCLIB EXEC C,L=11,N='AL68.PROCLIB',S='1,1,1' JCL PROCEDURES
//TESTS EXEC P,L=12,N='AL68.TESTS',S='20,1,4'   TESTS
//TESTJOB EXEC P,L=13,N='AL68.TESTJOB',S='20,1,4',Q=   TEST JOBS ETC
//INIT EXEC P,L=14,N='AL68.INIT',S='1,1,1' COMPILER INIT FILES
//ULIBENV EXEC P,L=15,N='AL68.USERLIB.ENV107',S='4,1,4' LIB ENV
//ZLIBTXT EXEC C,L=16,N='AL68.ZLIB.TXT127',S='4,2,2'
//ULIBTXT EXEC P,L=17,N='AL68.USERLIB.TXT107',S='4,2,2'
//XREFTXT EXEC P,L=18,N='AL68.XREF.SOURCE',S='1,1,1'
//PACKTXT EXEC P,L=19,N='AL68.PACKTXT',S='1,1,1' SRC OF PACK/UNPACK
//CHANGES EXEC F,L=20,N='AL68.CHANGES',S='1,1' DESCRIPTION OF CHANGES
//BUGS EXEC F,L=21,N='AL68.BUGS',S='1,1' BUG LIST
//UNIMPLEM EXEC F,L=22,N='AL68.UNIMPLEM',S='1,1'
//DIFFS EXEC F,L=23,N='AL68.DIFFS',S='1,1' DIFFERENCES FROM ALGOL 68
//PACKDOC EXEC F,L=24,N='AL68.PACKDOC',S='1,1' INFO FOR PACK/UNPACK
//CKJOBS EXEC C,L=25,N='AL68.MAKE.JOBS',S='2,1,4' COMPILE COMPILER
