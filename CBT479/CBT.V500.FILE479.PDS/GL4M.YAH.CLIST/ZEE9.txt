/* REXX */
TRACE ERR
CALL ZEEYC
IF RESULT = 4 THEN RETURN
MS = ''
"ISPEXEC VPUT (MS) PROFILE"
"ISPEXEC VGET (LT LF D1 D2 D3 D4 D5) PROFILE"
IF LF ¬= '' THEN DO
  MS = 'YOU HAVE ALREADY USED THE FULL HOUSE SLOT'
  "ISPEXEC VPUT (MS) PROFILE"
  RETURN
END
LF = 0
M1 = 0
IF D1 = D2 THEN DO
  IF D1 = D3 THEN IF D4 = D5 THEN DO
    IF D4 ¬= D1 THEN M1 = 3
  END
  IF D1 = D4 THEN IF D3 = D5 THEN DO
    IF D3 ¬= D1 THEN M1 = 3
  END
  IF D1 = D5 THEN IF D4 = D3 THEN DO
    IF D4 ¬= D1 THEN M1 = 3
  END
END
IF D1 = D3 THEN DO
  IF D1 = D4 THEN IF D2 = D5 THEN DO
    IF D2 ¬= D1 THEN M1 = 3
  END
  IF D1 = D5 THEN IF D2 = D4 THEN DO
    IF D2 ¬= D1 THEN M1 = 3
  END
END
IF D1 = D4 THEN DO
  IF D1 = D5 THEN IF D3 = D2 THEN DO
    IF D3 ¬= D1 THEN M1 = 3
  END
END
IF D2 = D3 THEN DO
  IF D2 = D4 THEN IF D1 = D5 THEN DO
    IF D2 ¬= D1 THEN M1 = 3
  END
  IF D2 = D5 THEN IF D1 = D4 THEN DO
    IF D2 ¬= D1 THEN M1 = 3
  END
END
IF D2 = D4 THEN DO
  IF D2 = D5 THEN IF D1 = D3 THEN DO
    IF D2 ¬= D1 THEN M1 = 3
  END
END
IF D3 = D4 THEN DO
  IF D4 = D5 THEN IF D1 = D2 THEN DO
    IF D3 ¬= D1 THEN M1 = 3
  END
END
IF M1 = 3 THEN LF = 25
IF M1 ¬= 3 THEN DO
  MS = 'AN EMPTY HOUSE'
  "ISPEXEC VPUT (MS) PROFILE"
END
LT = LT + LF
"ISPEXEC VPUT (LT LF) PROFILE"
CALL ZEELB
CALL ZEET
CALL ZEER
RETURN
