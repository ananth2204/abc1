/* REXX */
TRACE ERR
CALL ZEEUB
"ISPEXEC VGET (UT LT U1 U2 U3 U4 U5 U6) PROFILE"
"ISPEXEC VGET (L3 L4 LF LS LL LY LC LB) PROFILE"
SELECT
WHEN U1 = '' THEN  NOP
WHEN U2 = '' THEN  NOP
WHEN U3 = '' THEN  NOP
WHEN U4 = '' THEN  NOP
WHEN U5 = '' THEN  NOP
WHEN U6 = '' THEN  NOP
WHEN L3 = '' THEN  NOP
WHEN L4 = '' THEN  NOP
WHEN LF = '' THEN  NOP
WHEN LS = '' THEN  NOP
WHEN LL = '' THEN  NOP
WHEN LY = '' THEN  NOP
WHEN LC = '' THEN  NOP
OTHERWISE
  DO
    "ISPEXEC VGET (HS1 HS2 HS3 HS4 HS5) PROFILE"
    GT = LT + UT
    IF GT >= HS1 THEN DO
      HS5 = HS4
      HS4 = HS3
      HS3 = HS2
      HS2 = HS1
      HS1 = GT
    END
    IF GT < HS1 THEN IF GT >= HS2 THEN DO
      HS5 = HS4
      HS4 = HS3
      HS3 = HS2
      HS2 = GT
    END
    IF GT < HS2 THEN IF GT >= HS3 THEN DO
      HS5 = HS4
      HS4 = HS3
      HS3 = GT
    END
    IF GT < HS3 THEN IF GT >= HS4 THEN DO
      HS5 = HS4
      HS4 = GT
    END
    IF GT < HS4 THEN IF GT > HS5 THEN DO
      HS5 = GT
    END
    RS = 'YES'
    "ISPEXEC VPUT (HS1 HS2 HS3 HS4 HS5 RS) PROFILE"
    MS = 'Thanks for playing. Enter "99" to start a new game!'
    IF LT + UT > 300 THEN MS = 'GREAT GAME!' MS
    "ISPEXEC VPUT (MS) PROFILE"
  END
END
GT = LT + UT
"ISPEXEC VPUT (GT) PROFILE"
RETURN
