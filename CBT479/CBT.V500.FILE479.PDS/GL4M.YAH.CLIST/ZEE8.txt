/* REXX */
TRACE ERR
CALL ZEEYC
IF RESULT = 4 THEN RETURN
MS = ''
"ISPEXEC VPUT (MS) PROFILE"
"ISPEXEC VGET (LT L4 D1 D2 D3 D4 D5) PROFILE"
IF L4 ¬= '' THEN DO
  MS = 'YOU HAVE ALREADY USED THE 4 OF A KIND SLOT'
  "ISPEXEC VPUT (MS) PROFILE"
  RETURN
END
L4 = 0
M1 = 0
IF D1 = D2 THEN IF D1 = D3 THEN IF D1 = D4 THEN M1 = 3
IF D1 = D2 THEN IF D1 = D3 THEN IF D1 = D5 THEN M1 = 3
IF D1 = D2 THEN IF D1 = D4 THEN IF D1 = D5 THEN M1 = 3
IF D1 = D3 THEN IF D1 = D4 THEN IF D1 = D5 THEN M1 = 3
IF D2 = D3 THEN IF D2 = D4 THEN IF D2 = D5 THEN M1 = 3
IF M1 = 3 THEN L4 = D1 + D2 + D3 + D4 + D5
IF M1 ¬= 3 THEN DO
  MS = 'NO FOUR OF A KIND'
  "ISPEXEC VPUT (MS) PROFILE"
END
LT = LT + L4
"ISPEXEC VPUT (L4 LT) PROFILE"
CALL ZEELB
CALL ZEET
CALL ZEER
RETURN
