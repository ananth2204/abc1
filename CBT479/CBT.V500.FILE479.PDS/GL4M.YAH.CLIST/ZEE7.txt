/* REXX */
TRACE ERR
CALL ZEEYC
IF RESULT = 4 THEN RETURN
MS = ''
"ISPEXEC VPUT (MS) PROFILE"
"ISPEXEC VGET (LT L3 D1 D2 D3 D4 D5) PROFILE"
IF L3 ¬= '' THEN DO
  MS = 'YOU HAVE ALREADY USED THE 3 OF A KIND SLOT'
  "ISPEXEC VPUT (MS) PROFILE"
  RETURN
END
L3 = 0
M1 = 0
IF D1 = D2 THEN DO
  IF D1 = D3 THEN M1 = 3
  IF D1 = D4 THEN M1 = 3
  IF D1 = D5 THEN M1 = 3
END
IF D1 = D3 THEN DO
  IF D1 = D4 THEN M1 = 3
  IF D1 = D5 THEN M1 = 3
END
IF D1 = D4 THEN DO
  IF D1 = D5 THEN M1 = 3
END
IF D2 = D3 THEN DO
  IF D2 = D4 THEN M1 = 3
  IF D2 = D5 THEN M1 = 3
END
IF D2 = D4 THEN DO
  IF D2 = D5 THEN M1 = 3
END
IF D3 = D4 THEN DO
  IF D4 = D5 THEN M1 = 3
END
IF M1 = 3 THEN L3 = D1 + D2 + D3 + D4 + D5
IF M1 ¬= 3 THEN DO
  MS = 'NO THREE OF A KIND'
  "ISPEXEC VPUT (MS) PROFILE"
END
LT = LT + L3
"ISPEXEC VPUT (L3 LT) PROFILE"
CALL ZEELB
CALL ZEET
CALL ZEER
RETURN
